import matplotlib.pyplot as plt
import numpy as np
#Nicotine content data
nicotine_data=[1.09, 1.92, 2.31, 1.79, 2.28, 1.74, 1.47, 1.97, 0.85, 1.24, 1.58, 2.03, 1.70, 2.17, 2.55, 2.11, 1.86, 1.90, 1.68, 1.51, 1.64, 0.72, 1.69, 1.85, 1.82, 1.79, 2.46, 1.88, 2.08, 1.67, 1.37, 1.93, 1.40, 1.64, 2.09, 1.75, 1.63, 2.37, 1.75, 1.69]
#Create BOX Plot
plt.boxplot(nicotine_data)
plt.title('Box plot of Nicotine content')
plt.ylabel('nicotine content(mg)')
plt.show()
#Analyze the data
mean_nicotine=np.mean(nicotine_data)
median_nicotine=np.median(nicotine_data)
q1=np.percentile(nicotine_data,25)
q3=np.percentile(nicotine_data,75)
iqr=q3-q1
print(f'mean Nicotine content : {mean_nicotine}mg')
print(f'median Nicotine content : {median_nicotine}mg')
print(f'Interquartile Range(IQR): {iqr}mg')
print(f'Q1:{q1}mg')
print(f'Q3:{q3}mg')
#identify potential outliers
lower_bound=q1-1.5*iqr
upper_bound=q3+1.5*iqr
outliers=[x for x in nicotine_data if xupper_bound]
if len(outliers)>0:
print('Potential outliers:')
for outlier in outliers:
print(f'{outlier}mg')
else:
print('No potential outliers found')
